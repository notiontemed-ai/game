/**
 * Создает новый файл игровой сессии из текущего каталога.
 *
 * Поведение:
 * - создает новую таблицу;
 * - копирует из текущей таблицы листы START_SETUP / TECH_PARAMETERS, если они есть;
 * - если листов нет, создает их по шаблону;
 * - в новой таблице подготавливает START_SETUP значениями из модели.
 */
function createNewGameSessionFromCatalog() {
  const cfg = INITIALIZATION_CONFIG;

  const catalogSs = SpreadsheetApp.getActiveSpreadsheet();
  const ts = Utilities.formatDate(new Date(), Session.getScriptTimeZone() || 'Europe/Paris', 'yyyy-MM-dd HH:mm');
  const newSs = SpreadsheetApp.create(`Clinic Game Session ${ts}`);

  moveSpreadsheetToTargetFolder_(newSs, cfg.targetFolderId);

  const sheets = bootstrapRequiredSheetsInNewSession_(catalogSs, newSs, cfg);
  const startSheet = sheets.startSheet;
  const techSheet = sheets.techSheet;

  ensureTechParametersTemplateForInit_(techSheet, catalogSs);
  ensureStartSetupTemplateForInit_(startSheet, cfg.sourceMarkerText);
  applyLotBindingsInStartSetup_(newSs);

  cleanupDefaultSheetInNewSession_(newSs, cfg);

  // Заполняем только в новой книге (аналог fillStartSetupFromModel, но без привязки к active ss).
  fillStartSetupFromModelInSpreadsheet_(newSs);

  showSessionCreatedDialog_(newSs.getUrl());
}

function showSessionCreatedDialog_(url) {
  const safeUrl = String(url || '').replace(/"/g, '&quot;');
  const html = HtmlService.createHtmlOutput(
    '<div style="font-family:Arial,sans-serif;padding:12px;line-height:1.5">' +
      '<div style="margin-bottom:10px"><b>Создан новый файл игровой сессии.</b></div>' +
      '<div style="margin-bottom:10px"><a href="' + safeUrl + '" target="_blank">Ссылка</a></div>' +
      '<div>Далее откройте файл и запустите <b>Initialize session from START_SETUP</b>.</div>' +
    '</div>'
  ).setWidth(420).setHeight(180);

  SpreadsheetApp.getUi().showModalDialog(html, 'Game session created');
}


function cleanupDefaultSheetInNewSession_(ss, cfg) {
  const sheets = ss.getSheets();
  if (sheets.length <= 1) return;

  const protectedNames = {
    [cfg.startSheetName]: true,
    [cfg.techSheetName]: true,
    [cfg.lotListSheetName]: true,
    SESSION_CONFIG: true,
    TEAM_STATE: true,
    ROUND_RESULTS: true,
    ROUND_INPUT: true
  };

  for (let i = 0; i < sheets.length; i++) {
    const sh = sheets[i];
    const name = sh.getName();
    const looksDefault = /^Sheet\d*$/.test(name);
    if (!looksDefault) continue;
    if (protectedNames[name]) continue;
    if (ss.getSheets().length <= 1) return;
    ss.deleteSheet(sh);
    return;
  }
}

function moveSpreadsheetToTargetFolder_(ss, folderId) {
  if (!folderId) return;

  const file = DriveApp.getFileById(ss.getId());
  const targetFolder = DriveApp.getFolderById(folderId);
  file.moveTo(targetFolder);
}

}

function bootstrapRequiredSheetsInNewSession_(sourceSs, targetSs, cfg) {
  const allSheets = targetSs.getSheets();

  // Если есть только дефолтный лист, переиспользуем его как START_SETUP,
  // чтобы новая книга точно не оставалась "пустой".
  let startSheet = targetSs.getSheetByName(cfg.startSheetName);
  if (!startSheet && allSheets.length === 1 && isDefaultSheetName_(allSheets[0].getName())) {
    allSheets[0].setName(cfg.startSheetName);
    startSheet = allSheets[0];
  }
  if (!startSheet) {
    startSheet = getOrCreateSheetForInit_(targetSs, cfg.startSheetName);
  }

  let techSheet = targetSs.getSheetByName(cfg.techSheetName);
  if (!techSheet) {
    techSheet = getOrCreateSheetForInit_(targetSs, cfg.techSheetName);
  }

  const lotSheet = getOrCreateSheetForInit_(targetSs, cfg.lotListSheetName);

  const copiedStart = copySheetValuesByName_(sourceSs, startSheet, cfg.startSheetName);
  const copiedTech = copySheetValuesByName_(sourceSs, techSheet, cfg.techSheetName);
  const copiedLot = copySheetValuesByName_(sourceSs, lotSheet, cfg.lotListSheetName);

  if (!copiedStart) {
    ensureStartSetupTemplateForInit_(startSheet, cfg.sourceMarkerText);
  }
  if (!copiedTech) {
    ensureTechParametersTemplateForInit_(techSheet, sourceSs);
  }
  if (!copiedLot) {
    ensureLotListTemplateForInit_(lotSheet);
  }

  return { startSheet, techSheet };
}

function copySheetValuesByName_(sourceSs, targetSheet, sourceSheetName) {
  const src = sourceSs.getSheetByName(sourceSheetName);
  if (!src) return false;

  const values = src.getDataRange().getValues();
  targetSheet.clear();

  if (!values || values.length === 0) return true;
  targetSheet.getRange(1, 1, values.length, values[0].length).setValues(values);

  return true;
}

function isDefaultSheetName_(name) {
  return /^Sheet\d*$/.test(name) || /^Лист\d*$/.test(name);
}

/**
 * Опциональный helper, если нужно проставить формулы в START_SETUP.
 * ВАЖНО: разделитель аргументов в формулах — ";".
 *
 * Пример использования:
 * setStartSetupFormulaByCode_('selected_lot_name', '=IFERROR(VLOOKUP(D2;LOT_DIRECTORY!A:Z;2;FALSE);"")');
 */
function setStartSetupFormulaByCode_(parameterCode, formulaA1) {
  const cfg = INITIALIZATION_CONFIG;
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const startSheet = mustGetSheetForInit_(ss, cfg.startSheetName);

  const values = startSheet.getDataRange().getValues();
  if (values.length < 2) throw new Error('START_SETUP пуст.');

  const headers = values[0].map(v => String(v || '').trim());
  const h = buildHeaderMapForInit_(headers);

  const colParameterCode = pickHeaderIndex_(h, ['parameter_code'], 1);
  const colValueInput = pickHeaderIndex_(h, ['value_input'], 3);

  let targetRow = null;
  for (let r = 1; r < values.length; r++) {
    const code = String(values[r][colParameterCode] || '').trim();
    if (code === parameterCode) {
      targetRow = r + 1;
      break;
    }
  }

  if (!targetRow) {
    throw new Error(`В START_SETUP не найден parameter_code=${parameterCode}`);
  }

  // Проверка на случайно переданный запятый разделитель.
  if (formulaA1.indexOf(',') >= 0 && formulaA1.indexOf(';') < 0) {
    throw new Error('Формула содержит ",". Для Google Sheets используйте ";" как разделитель аргументов.');
  }

  startSheet.getRange(targetRow, colValueInput + 1).setFormula(formulaA1);
}

function readTechParametersMapForInit_(techSheet) {
  const values = techSheet.getDataRange().getValues();
  if (values.length < 2) return {};

  const headers = values[0].map(v => String(v || '').trim());
  const h = buildHeaderMapForInit_(headers);

  const colCode = pickHeaderIndex_(h, ['parameter_code'], 1);
  const colValue = pickHeaderIndex_(h, ['value'], 2);

  const map = {};
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    const code = String(row[colCode] || '').trim();
    if (!code) continue;
    map[code] = row[colValue];
  }
  return map;
}

function resolveModelValueByCode_(techMap, code) {
  if (!techMap) return null;

  // Прямое совпадение
  if (code in techMap) return techMap[code];

  // Частые алиасы для start_cash_balance
  if (code === 'start_cash_balance') {
    if ('default_start_cash_balance' in techMap) return techMap.default_start_cash_balance;
    if ('starting_cash_balance' in techMap) return techMap.starting_cash_balance;
  }

  return null;
}

function buildStartSetupRowByCode_(startValues, colParameterCode) {
  const map = {};
  for (let r = 1; r < startValues.length; r++) {
    const code = String(startValues[r][colParameterCode] || '').trim();
    if (!code) continue;
    map[code] = r + 1;
  }
  return map;
}

function isModelPulledMarker_(source, marker) {
  return String(source || '').toLowerCase().indexOf(String(marker || '').toLowerCase()) >= 0;
}

function applyLotBindingsInStartSetup_(ss) {
  const cfg = INITIALIZATION_CONFIG;
  const startSheet = ss.getSheetByName(cfg.startSheetName);
  const lotSheet = ss.getSheetByName(cfg.lotListSheetName);
  if (!startSheet || !lotSheet) return;

  // D3 — value_input для selected_lot_id в шаблоне START_SETUP.
  const lotIdCell = startSheet.getRange(3, 4);
  const rule = SpreadsheetApp.newDataValidation()
    .requireValueInRange(lotSheet.getRange('A2:A'), true)
    .setAllowInvalid(false)
    .build();
  lotIdCell.setDataValidation(rule);
}

function buildTechRowsFromCoefficients_(ss) {
  if (!ss) return null;

  const cfg = INITIALIZATION_CONFIG;
  const coeffSheet = ss.getSheetByName(cfg.coefficientsSheetName);
  if (!coeffSheet) return null;

  const values = coeffSheet.getDataRange().getValues();
  if (!values || values.length < 2) return null;

  const headers = values[0].map(v => String(v || '').trim());
  const hm = buildHeaderMapForInit_(headers);
  const hmLower = buildHeaderMapForInit_(headers.map(h => String(h).toLowerCase()));

  const codeIdx = pickHeaderIndex_(hm, ['parameter_code','code','Код'],
    pickHeaderIndex_(hmLower, ['parameter_code','code','код'], 1));
  const blockIdx = pickHeaderIndex_(hm, ['block','Блок'],
    pickHeaderIndex_(hmLower, ['block','блок'], 0));
  const valueIdx = pickHeaderIndex_(hm, [cfg.coefficientsExampleValueHeader, 'Значение (input)', 'value'],
    pickHeaderIndex_(hmLower, [String(cfg.coefficientsExampleValueHeader).toLowerCase(), 'значение (input)', 'value'], 4));

  const rows = [['block', 'parameter_code', 'value', 'comment']];
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    const code = String(row[codeIdx] || '').trim();
    if (!code) continue;
    const block = String(row[blockIdx] || 'COEFF').trim() || 'COEFF';
    const v = row[valueIdx];
    if (v === '' || v === null || v === undefined) continue;
    rows.push([block, code, v, 'Imported from 08_Coefficients / Пример значения (input)']);
  }

  return rows.length > 1 ? rows : null;
}



function buildConstantsMap_(ss) {
  if (!ss) return {};
  const cfg = INITIALIZATION_CONFIG;
  const sheet = ss.getSheetByName(cfg.constantsSheetName);
  if (!sheet) return {};

  const values = sheet.getDataRange().getValues();
  if (!values || values.length < 2) return {};

  const headers = values[0].map(v => String(v || '').trim());
  const hm = buildHeaderMapForInit_(headers);
  const hmLower = buildHeaderMapForInit_(headers.map(h => String(h).toLowerCase()));

  const codeIdx = pickHeaderIndex_(hm, ['parameter_code','code','Код'],
    pickHeaderIndex_(hmLower, ['parameter_code','code','код'], 1));
  const valueIdx = pickHeaderIndex_(hm, [cfg.constantsValueHeader, 'value', 'Значение'],
    pickHeaderIndex_(hmLower, [String(cfg.constantsValueHeader).toLowerCase(), 'value', 'значение'], 4));

  const map = {};
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    const code = String(row[codeIdx] || '').trim();
    if (!code) continue;
    const v = row[valueIdx];
    if (v === '' || v === null || v === undefined) continue;
    map[code] = v;
  }
  return map;
}

function buildHiddenStateStartMap_(ss) {
  if (!ss) return {};
  const cfg = INITIALIZATION_CONFIG;
  const sheet = ss.getSheetByName(cfg.hiddenStateSheetName);
  if (!sheet) return {};

  const values = sheet.getDataRange().getValues();
  if (!values || values.length < 2) return {};

  const headers = values[0].map(v => String(v || '').trim());
  const hm = buildHeaderMapForInit_(headers);
  const hmLower = buildHeaderMapForInit_(headers.map(h => String(h).toLowerCase()));

  const codeIdx = pickHeaderIndex_(hm, ['parameter_code','code','Код'],
    pickHeaderIndex_(hmLower, ['parameter_code','code','код'], 1));
  const startIdx = pickHeaderIndex_(hm, [cfg.hiddenStateStartValueHeader, 'start_value', 'Стартовые значения'],
    pickHeaderIndex_(hmLower, [String(cfg.hiddenStateStartValueHeader).toLowerCase(), 'start_value', 'стартовые значения'], 3));

  const map = {};
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    const code = String(row[codeIdx] || '').trim();
    if (!code) continue;
    const v = row[startIdx];
    if (v === '' || v === null || v === undefined) continue;
    map[code] = v;
  }

  return map;
}


function getCoefficientValueByCode_(ss, codeToFind) {
  if (!ss) return null;
  const cfg = INITIALIZATION_CONFIG;
  const sheet = ss.getSheetByName(cfg.coefficientsSheetName);
  if (!sheet) return null;

  const values = sheet.getDataRange().getValues();
  if (!values || values.length < 2) return null;

  const headers = values[0].map(v => String(v || '').trim());
  const hm = buildHeaderMapForInit_(headers);
  const hmLower = buildHeaderMapForInit_(headers.map(h => String(h).toLowerCase()));

  const codeIdx = pickHeaderIndex_(hm, ['parameter_code','code','Код'],
    pickHeaderIndex_(hmLower, ['parameter_code','code','код'], 1));
  const valueIdx = pickHeaderIndex_(hm, [cfg.coefficientsExampleValueHeader, 'Значение (input)', 'value'],
    pickHeaderIndex_(hmLower, [String(cfg.coefficientsExampleValueHeader).toLowerCase(), 'значение (input)', 'value'], 4));

  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    const code = String(row[codeIdx] || '').trim();
    if (code !== codeToFind) continue;
    const v = row[valueIdx];
    if (v === '' || v === null || v === undefined) return null;
    return v;
  }

  return null;
}

function getOrCreateSheetForInit_(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function mustGetSheetForInit_(ss, name) {
  const sheet = ss.getSheetByName(name);
  if (!sheet) throw new Error(`Не найден лист "${name}"`);
  return sheet;
}

function buildHeaderMapForInit_(headers) {
  const map = {};
  headers.forEach((v, i) => {
    map[String(v || '').trim()] = i;
  });
  return map;
}

function pickHeaderIndex_(map, variants, fallbackIndex) {
  for (let i = 0; i < variants.length; i++) {
    const key = variants[i];
    if (key in map) return map[key];
  }
  return fallbackIndex;
}

function unique_(arr) {
  const s = {};
  (arr || []).forEach(v => {
    s[String(v)] = true;
  });
  return Object.keys(s);
}
